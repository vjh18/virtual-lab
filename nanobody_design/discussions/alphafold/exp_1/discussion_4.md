import sys
import pandas as pd
from Bio.PDB import PDBParser, Structure
from typing import List, Tuple

def parse_pLDDT_from_pdb(pdb_file: str) -> float:
    """
    Parse the pLDDT score from a PDB file.

    Args:
        pdb_file (str): Path to the PDB file.

    Returns:
        float: Average pLDDT score for the protein-protein interface.
    """
    parser = PDBParser(QUIET=True)
    structure: Structure = parser.get_structure('complex', pdb_file)

    interface_plddts: List[float] = []
    
    for model in structure:
        for chain in model:
            for residue in chain:
                for atom in residue:
                    # Assuming B-factor holds pLDDT scores
                    interface_plddts.append(atom.get_bfactor())

    # Compute the average pLDDT for the interface
    average_plddt: float = sum(interface_plddts) / len(interface_plddts) if interface_plddts else 0.0
    return average_plddt
